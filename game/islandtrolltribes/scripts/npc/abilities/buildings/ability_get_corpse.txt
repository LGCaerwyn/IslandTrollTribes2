// Picks up a nearby corpse.
// Can be autocast which will continuously pick up meat around the building until at full capacity or until no more corpses remain.
"ability_get_corpse"
{
    "BaseClass"             "ability_datadriven"
    "AbilityTextureName"    "ability_building_getcorpse"
    "AbilityBehavior"       "DOTA_ABILITY_BEHAVIOR_NO_TARGET | DOTA_ABILITY_BEHAVIOR_AUTOCAST"
    "MaxLevel"              "1"
    "AbilityCastRange"      "300"

    "OnSpellStart"
    {
        "RunScript"
        {
            "ScriptFile"    "buildings/get_corpse.lua"
            "Function"      "GetCorpses"
        }
    }

    "Modifiers"
    {
        "modifier_get_corpses"
        {
            "Passive"           "1"
            "IsHidden"          "1"
    
            "OnCreated"
            {
                "RunScript"
                {
                    "ScriptFile"    "mechanics/autocast.lua"
                    "Function"      "AutocastOn"
                }

                "RunScript"
                {
                    "ScriptFile"    "buildings/get_corpse.lua"
                    "Function"      "FillSlots"
                }
            }

            "ThinkInterval"  "1.0"
            "OnIntervalThink"
            {
                "RunScript"
                {
                    "ScriptFile"    "buildings/get_corpse.lua"
                    "Function"      "GetCorpseAutocast"
                }
            }
        }
    }
}
